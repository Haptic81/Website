local moonGravityEnabled = false
local gravityConnection

local function resetGravity()
    local char = Players.LocalPlayer.Character
    if char then
        local humanoid = char:FindFirstChildOfClass("Humanoid")
        if humanoid then humanoid.JumpPower = 50 end
    end
end

gravityButton.MouseButton1Click:Connect(function()
    moonGravityEnabled = not moonGravityEnabled
    if moonGravityEnabled then
        gravityButton.BackgroundColor3 = Color3.fromRGB(80, 80, 80)
        local char = Players.LocalPlayer.Character
        if char then
            local humanoid = char:FindFirstChildOfClass("Humanoid")
            local root = char:FindFirstChild("HumanoidRootPart")
            if humanoid and root then
                humanoid.UseJumpPower = true
                humanoid.JumpPower = 96
                if gravityConnection then gravityConnection:Disconnect() end
                gravityConnection = RunService.Stepped:Connect(function()
                    if not moonGravityEnabled then return end
                    if root.Velocity.Y < 0 then
                        root.Velocity = Vector3.new(root.Velocity.X, root.Velocity.Y * 0.9, root.Velocity.Z)
                    end
                end)
            end
        end
    else
        gravityButton.BackgroundColor3 = Color3.fromRGB(51, 51, 51)
        if gravityConnection then gravityConnection:Disconnect() end
        gravityConnection = nil
        resetGravity()
    end
end)

local teleportUp = true
teleportButton.MouseButton1Click:Connect(function()
    local char = Players.LocalPlayer.Character
    if char then
        local root = char:FindFirstChild("HumanoidRootPart")
        if root then
            if teleportUp then
                root.CFrame = root.CFrame + Vector3.new(0, 150, 0)
                teleportButton.Text = "Teleport Down"
            else
                root.CFrame = root.CFrame - Vector3.new(0, 189, 0)
                teleportButton.Text = "Teleport Up"
            end
            teleportUp = not teleportUp
        end
    end
end)

local speedEnabled = false
local speedConnection
local speedValue = 60

speedButton.MouseButton1Click:Connect(function()
    speedEnabled = not speedEnabled
    if speedEnabled then
        speedButton.BackgroundColor3 = Color3.fromRGB(80, 80, 80)
        local player = Players.LocalPlayer
        local char = player.Character or player.CharacterAdded:Wait()
        local root = char:WaitForChild("HumanoidRootPart")
        local camera = workspace.CurrentCamera
        if speedConnection then speedConnection:Disconnect() end
        speedConnection = RunService.RenderStepped:Connect(function()
            if not speedEnabled then return end
            local mv = Vector3.zero
            if UserInputService:IsKeyDown(Enum.KeyCode.W) then mv += camera.CFrame.LookVector end
            if UserInputService:IsKeyDown(Enum.KeyCode.S) then mv -= camera.CFrame.LookVector end
            if UserInputService:IsKeyDown(Enum.KeyCode.A) then mv -= camera.CFrame.RightVector end
            if UserInputService:IsKeyDown(Enum.KeyCode.D) then mv += camera.CFrame.RightVector end
            if mv.Magnitude > 0 then
                mv = mv.Unit * speedValue
            else
                mv = Vector3.zero
            end
            root.AssemblyLinearVelocity = Vector3.new(mv.X, root.AssemblyLinearVelocity.Y, mv.Z)
        end)
    else
        speedButton.BackgroundColor3 = Color3.fromRGB(51, 51, 51)
        if speedConnection then speedConnection:Disconnect() end
        speedConnection = nil
    end
end)

local espEnabled = false
local espBoxes = {}

local function createBox(player)
    local box = {}
    box.parts = {}
    local sizeScale = Vector3.new(1,1,1)
    local function createAdornment(partName, size, color)
        local adorn = Instance.new("BoxHandleAdornment")
        adorn.Name = partName
        adorn.Adornee = nil
        adorn.AlwaysOnTop = true
        adorn.ZIndex = 10
        adorn.Transparency = 0.4
        adorn.Color3 = color
        adorn.Size = size
        adorn.Parent = workspace
        return adorn
    end
    local color = Color3.new(1,1,1)
    box.parts.Head = createAdornment("Head", Vector3.new(2, 1, 1), color)
    box.parts.Torso = createAdornment("Torso", Vector3.new(2, 2, 1), color)
    box.parts.LeftArm = createAdornment("LeftArm", Vector3.new(1, 2, 1), color)
    box.parts.RightArm = createAdornment("RightArm", Vector3.new(1, 2, 1), color)
    box.parts.LeftLeg = createAdornment("LeftLeg", Vector3.new(1, 2, 1), color)
    box.parts.RightLeg = createAdornment("RightLeg", Vector3.new(1, 2, 1), color)
    return box
end

local function updateBox(box)
    local char = box.player.Character
    if not char then return end
    local head = char:FindFirstChild("Head")
    local torso = char:FindFirstChild("UpperTorso") or char:FindFirstChild("Torso")
    local leftArm = char:FindFirstChild("LeftUpperArm") or char:FindFirstChild("Left Arm")
    local rightArm = char:FindFirstChild("RightUpperArm") or char:FindFirstChild("Right Arm")
    local leftLeg = char:FindFirstChild("LeftUpperLeg") or char:FindFirstChild("Left Leg")
    local rightLeg = char:FindFirstChild("RightUpperLeg") or char:FindFirstChild("Right Leg")
    if not head or not torso or not leftArm or not rightArm or not leftLeg or not rightLeg then return end
    box.parts.Head.Adornee = head
    box.parts.Torso.Adornee = torso
    box.parts.LeftArm.Adornee = leftArm
    box.parts.RightArm.Adornee = rightArm
    box.parts.LeftLeg.Adornee = leftLeg
    box.parts.RightLeg.Adornee = rightLeg
end

local function clearBoxes()
    for _, box in pairs(espBoxes) do
        for _, part in pairs(box.parts) do
            part:Destroy()
        end
    end
    espBoxes = {}
end

local function removeBox(player)
    local box = espBoxes[player]
    if box then
        for _, part in pairs(box.parts) do
            part:Destroy()
        end
        espBoxes[player] = nil
    end
end

Players.PlayerRemoving:Connect(removeBox)

espButton.MouseButton1Click:Connect(function()
    espEnabled = not espEnabled
    if espEnabled then
        espButton.BackgroundColor3 = Color3.fromRGB(80, 80, 80)
        for _, player in pairs(Players:GetPlayers()) do
            if player ~= Players.LocalPlayer then
                removeBox(player)
                local box = createBox(player)
                box.player = player
                espBoxes[player] = box
            end
        end
        spawn(function()
            while espEnabled do
                for player, box in pairs(espBoxes) do
                    if player.Character then
                        updateBox(box)
                    else
                        removeBox(player)
                    end
                end
                RunService.Heartbeat:Wait()
            end
            clearBoxes()
        end)
    else
        espButton.BackgroundColor3 = Color3.fromRGB(51, 51, 51)
        clearBoxes()
    end
end)
